Updated Code:

ADD GLOBAL VARIABLE:

uint32_t is_running = 0;

ADD FUNCTION:

void process_input()
{
	SDL_Event event;
	SDL_PollEvent(&event);
	switch(event.type)
	{
		case SDL_QUIT:
		{
			is_running = 0;
		} break;
	}
}

EDIT FUNCTION:

int main()
{
	sdl_init();

	is_running = 1;
	while(is_running)
	{
		process_input();
	}
	
	SDL_Delay(1000);

	return 0;
}
